\section{Evaluation Protocol}
\label{section:test-report}
\begin{itemize}
	\item \textbf{Baseline (CCD-ITI only):} Train on available CCD-ITI segments; report RT MAE/MSE and accuracy/F1.
	\item \textbf{Transfer (SuS \textrightarrow{} CCD-ITI):} Pretrain on SuS; fine-tune on CCD-ITI; compare metrics against baseline.
	\item \textbf{Verification:} Visualize SNR spectra and evoked responses to confirm signal components.
\end{itemize}
